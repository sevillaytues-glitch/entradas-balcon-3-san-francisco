📄 Codigo Google Apps Script necesario ▶ Mostrar

Copia este codigo en Extensions > Apps Script dentro de tu Google Sheet, despliégalo como Web App (acceso: "Cualquier persona") y pega la URL en el campo de arriba.

function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName("Entradas"); var params = JSON.parse(e.postData.contents); var action = params.action; if (action === "upsert") { var ticket = params.ticket; var data = sheet.getDataRange().getValues(); var headers = data[0]; var idCol = headers.indexOf("ID"); var found = false; for (var r = 1; r < data.length; r++) { if (data[r][idCol] === ticket.id) { // Update status and timestamp var statusCol = headers.indexOf("Estado"); var tsCol = headers.indexOf("FechaAcceso"); sheet.getRange(r+1, statusCol+1).setValue(ticket.status); if (tsCol >= 0) sheet.getRange(r+1, tsCol+1).setValue(ticket.accessedAt || ""); found = true; break; } } if (!found) { sheet.appendRow([ ticket.id, ticket.name, ticket.company, ticket.type, ticket.session, ticket.price, ticket.status, ticket.createdAt, ticket.accessedAt||"" ]); } return ContentService.createTextOutput( JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } if (action === "markAccessed") { var id = params.id; var ts = params.accessedAt; var d = sheet.getDataRange().getValues(); var h = d[0]; var iC = h.indexOf("ID"); var sC = h.indexOf("Estado"); var tC = h.indexOf("FechaAcceso"); for (var i = 1; i < d.length; i++) { if (d[i][iC] === id) { sheet.getRange(i+1, sC+1).setValue("Accedido"); if (tC >= 0) sheet.getRange(i+1, tC+1).setValue(ts); return ContentService.createTextOutput( JSON.stringify({ok:true,name:d[i][h.indexOf("Nombre")]})) .setMimeType(ContentService.MimeType.JSON); } } return ContentService.createTextOutput( JSON.stringify({ok:false,error:"not_found"})) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput( JSON.stringify({ok:false,error:"unknown_action"})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet() .getSheetByName("Entradas"); var data = sheet.getDataRange().getValues(); var h = data[0]; var rows = []; for (var i = 1; i < data.length; i++) { var row = {}; for (var j = 0; j < h.length; j++) row[h[j]] = data[i][j]; rows.push(row); } return ContentService.createTextOutput(JSON.stringify(rows)) .setMimeType(ContentService.MimeType.JSON); }
📚 Guia de configuracion
  1. Crea una Google Sheet con las columnas:
    ID, Nombre, Empresa, Tipo, Sesion, Precio, Estado, FechaCreacion, FechaAcceso
  2. Comparte la hoja como "Cualquier persona con el enlace puede ver".
  3. Crea una API Key en Google Cloud Console con acceso a la Sheets API v4.
  4. Copia el script de la seccion de abajo en Extensions > Apps Script, despliégalo como Web App con acceso "Cualquier persona".
  5. Pega las credenciales en los campos de la izquierda y pulsa "Guardar y conectar".
🔒 Las credenciales se guardan solo en el localStorage de este navegador y nunca se envian a terceros.
